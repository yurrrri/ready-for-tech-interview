### 1. stack 2개를 사용하여 Queue를 구현해보시오.

```swift
struct Queue<T> {
  var instack:[T] = []
  var outstack:[T] = []
  
  mutating func push(_ element: T) {
    instack.append(element)
  }
  
  mutating func pop() -> T? {
    if outstack.isEmpty {
      oustack = instack.reversed()
      instack.removeAll()
    }
    return outstack.popLast()
 }
}
 
```
